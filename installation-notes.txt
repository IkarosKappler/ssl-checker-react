
### Installation
https://www.typescriptlang.org/docs/handbook/react-&-webpack.html

## Folder Structure
-+
 |
 +-dist
 |
 +-src
   |
   +-components

## Create package.json
$ npm init -y



## Install webpack
$ npm install --save-dev webpack webpack-cli

## Install react and type files
$ npm install --save react react-dom
$ npm install --save-dev @types/react @types/react-dom

## Install ts-loader and source-map-loader
$ npm install --save-dev typescript ts-loader source-map-loader


## Create your default tsconfig.json file
~~~json
 {
    "compilerOptions": {
        "outDir": "./dist/",
        "sourceMap": true,
        "noImplicitAny": true,
        "module": "commonjs",
        "target": "es6",
        "jsx": "react"
    }
 }
~~~

## Create your first test component
File: src/components/Hello.tsx
~~~
 import * as React from "react";

 export interface HelloProps { compiler: string; framework: string; }

 export const Hello = (props: HelloProps) => <h1>Hello from {props.compiler} and {props.framework}!</h1>;
~~~

### Variante  with class compoents
~~~
 import * as React from "react";

 export interface HelloProps { compiler: string; framework: string; }

 // 'HelloProps' describes the shape of props.
 // State is never set so we use the '{}' type.
 export class Hello extends React.Component<HelloProps, {}> {
    render() {
        return <h1>Hello from {this.props.compiler} and {this.props.framework}!</h1>;
    }
 }
~~~


## Create your index.tsx file
File: src/index.tsx
~~~
 import * as React from "react";
 import * as ReactDOM from "react-dom";

 import { Hello } from "./components/Hello";

 ReactDOM.render(
    <Hello compiler="TypeScript" framework="React" />,
    document.getElementById("example")
 );
~~~


## Create the index.html main file
File: ./index.html
~~~
 <!DOCTYPE html>
 <html>
    <head>
        <meta charset="UTF-8" />
        <title>Hello React!</title>
    </head>
    <body>
        <div id="example"></div>

        <!-- Dependencies -->
        <script src="./node_modules/react/umd/react.development.js"></script>
        <script src="./node_modules/react-dom/umd/react-dom.development.js"></script>

        <!-- Main -->
        <script src="./dist/main.js"></script>
    </body>
 </html>
~~~



## Create the webpack config file
File: ./webpack.config.js
~~~
 module.exports = {
    mode: "production",

    // Enable sourcemaps for debugging webpack's output.
    devtool: "source-map",

    resolve: {
        // Add '.ts' and '.tsx' as resolvable extensions.
        extensions: [".ts", ".tsx"]
    },

    module: {
        rules: [
            {
                test: /\.ts(x?)$/,
                exclude: /node_modules/,
                use: [
                    {
                        loader: "ts-loader"
                    }
                ]
            },
            // All output '.js' files will have any sourcemaps re-processed by 'source-map-loader'.
            {
                enforce: "pre",
                test: /\.js$/,
                loader: "source-map-loader"
            }
        ]
    },

    // When importing a module whose path matches one of the following, just
    // assume a corresponding global variable exists and use that instead.
    // This is important because it allows us to avoid bundling all of our
    // dependencies, which allows browsers to cache those libraries between builds.
    externals: {
        "react": "React",
        "react-dom": "ReactDOM"
    }
 };
~~~
